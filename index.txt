        // --- CORE LOGIC ---
        let isOpened = false;
        
        function openGift() {
            if (isOpened) return;
            isOpened = true;

            // ·∫®n h·ªôp qu√†, hi·ªán card
            document.getElementById('giftSection').style.display = 'none';

        // --- TYPING EFFECT ---
        let msgIndex = 0;
        let charIndex = 0;
        const msgElement = document.getElementById('message');

        function typeWriter() {
            if (msgIndex < config.messages.length) {
                if (charIndex < config.messages[msgIndex].length) {
                    msgElement.innerHTML += config.messages[msgIndex].charAt(charIndex);
                    charIndex++;
                    setTimeout(typeWriter, config.typingSpeed);
                } else {
                    msgElement.innerHTML += "<br>"; // Xu·ªëng d√≤ng
                    msgIndex++;
                    charIndex = 0;
                    setTimeout(typeWriter, 300); // Delay gi·ªØa c√°c d√≤ng
                }
            }
        }

        // --- CANVAS PARTICLE SYSTEM (Tr√°i tim bay bay) ---
        const canvas = document.getElementById('canvas');
        const ctx = canvas.getContext('2d');
        let width, height;
        let particles = [];

        function resize() {
            width = canvas.width = window.innerWidth;
            height = canvas.height = window.innerHeight;
        }
        window.addEventListener('resize', resize);
        resize();

        class Particle {
            constructor() {
                this.x = Math.random() * width;
                this.y = height + Math.random() * 100;
                this.velY = Math.random() * -3 - 1; // Bay l√™n
                this.velX = Math.random() * 2 - 1;  // L·∫Øc l∆∞ ngang
                this.size = Math.random() * 15 + 10;
                this.color = `hsl(${Math.random() * 60 + 330}, 100%, 70%)`; // T√¥ng m√†u h·ªìng/ƒë·ªè
                this.rotation = Math.random() * Math.PI * 2;
            }

            update() {
                this.y += this.velY;
                this.x += Math.sin(this.y / 50) * 1; // Hi·ªáu ·ª©ng l·∫Øc l∆∞
                this.rotation += 0.05;
                
                // Reset n·∫øu bay qu√° m√†n h√¨nh
                if (this.y < -50) {
                    this.y = height + 50;
                    this.x = Math.random() * width;
                }
            }

            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.rotation);
                
                // V·∫Ω h√¨nh tr√°i tim
                ctx.beginPath();
                ctx.fillStyle = this.color;
                const topCurveHeight = this.size * 0.3;
                ctx.moveTo(0, topCurveHeight);
                ctx.bezierCurveTo(0, 0, -this.size / 2, 0, -this.size / 2, topCurveHeight);
                ctx.bezierCurveTo(-this.size / 2, (this.size + topCurveHeight) / 2, 0, this.size, 0, this.size);
                ctx.bezierCurveTo(0, this.size, this.size / 2, (this.size + topCurveHeight) / 2, this.size / 2, topCurveHeight);
                ctx.bezierCurveTo(this.size / 2, 0, 0, 0, 0, topCurveHeight);
                ctx.fill();
                ctx.closePath();
                
                ctx.restore();
            }
        }

        function initParticles() {
            // T·∫°o 50 h·∫°t tr√°i tim
            for (let i = 0; i < 50; i++) {
                particles.push(new Particle());
            }
            animate();
        }

        function animate() {
            ctx.clearRect(0, 0, width, height);
            particles.forEach(p => {
                p.update();
                p.draw();
            });
            requestAnimationFrame(animate);
        }
